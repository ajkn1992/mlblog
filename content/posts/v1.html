for alg in MLA:
    reg = alg()
    name = reg.__class__.__name__

    reg.fit(X_train, y_train)
    scores = cross_val_score(reg, X_train, y_train, cv=10)
    print('%s CV accuracy: %.3f +/- %.3f' % (name, np.mean(scores), np.std(scores)))
    
    print("Model Name {}: Train {}: Test: {}".format(name, reg.score(X_train, y_train), reg.score(X_test, y_test)))
    title = "Learning Curves ({})".format(name)
    cv = 10
    plot_learning_curve(reg, title, X_train, y_train, cv=cv)

class FamilySize(BaseEstimator, TransformerMixin):
    def __init__(self):
        pass
    
    def fit(self, X):
        return self  # nothing else to do
    
    def transform(self, X):
        # deep copy the df because of transform
        df = X.copy()

        df['FamilySize'] = df['SibSp'] + df['Parch']
        df.drop('SibSp',axis=1,inplace=True)
        df.drop('Parch',axis=1,inplace=True)
        df.drop(['Ticket','Cabin'], axis=1, inplace=True)
        
        return df
